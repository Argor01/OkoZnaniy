```typescript
  getMockClaimById,
  getMockDirectorCommunications,
  getMockResponseTemplates,
  getMockClaimsStatistics
} from '../mocks/claimsData';

// Получить все обращения
const allClaims = getMockClaims();

// Получить только новые обращения
const newClaims = getMockClaimsByStatus(ClaimStatus.NEW);

// Получить конкретное обращение
const claim = getMockClaimById(1);

// Получить коммуникации с дирекцией
const directorComms = getMockDirectorCommunications();

// Получить шаблоны ответов
const templates = getMockResponseTemplates();

// Получить статистику

```

## Особенности данных

1. **Реалистичные сценарии**: все обращения основаны на реальных ситуациях
2. **Разнообразие**: покрыты все типы и приоритеты обращений
3. **История**: у каждого обращения есть история изменения статусов
4. **Сообщения**: некоторые обращения содержат переписку
5. **Связи**: обращения связаны с заказами и пользователями
6. **Временные метки**: реалистичные даты и время создания

## Обновление данных

Для добавления новых тестовых обращений:
1. Добавьте новый объект в массив `mockClaims`
2. Обновите статистику в `mockClaimsStatistics`
3. При необходимости добавьте новых пользователей или заказы
4. Обновите тренд данных

## Примечания

- Все даты указаны в формате ISO 8601
- ID обращений уникальны и последовательны
- Внутренние заметки администраторов помечены флагом `isInternal: true`
- Непрочитанные сообщения от директора учитываются в `unreadCount`
